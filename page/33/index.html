slax 定制的脚本runme.sh

#!/bin/bash
#
# run this script to create a LiveCD in /tmp/livecd.iso
# Your kernel image has to be in $ROOT/boot/vmlinuz or $ROOT/vmlinuz
#

export PATH=.:./tools:../tools:/usr/sbin:/usr/bin:/sbin:/bin:/

# 确定在linux-live目录
CHANGEDIR=”dirnamereadlink -f $0`"<br />echo "Changing current directory to $CHANGEDIR"<br />cd $CHANGEDIR<br /><br /># <br />. liblinuxlive || exit 1<br />. config || exit 1<br /><br /># 复制所需程序到/usr/sbin /usr/lib 见 ./install 文件<br />./install $ROOT<br /><br /># 内核<br />VMLINUZ=$ROOT/boot/vmlinuz<br />if [ -L "$VMLINUZ" ]; then VMLINUZ=readlink -f $VMLINUZ; fi<br />if [ "ls $VMLINUZ 2>/dev/null`” = “” ]; then echo “cannot find $VMLINUZ”; exit 1; fi

header “Creating LiveCD from your Linux”

echo “Using kernel from $VMLINUZ”
echo “Using kernel modules from /lib/modules/$KERNEL”

# 制作initrd 镜像
echo “creating initrd image…”
cd initrd
./initrd_create
if [ “$?” -ne 0 ]; then exit; fi
cd ..

# 建立光盘目录和复制文件
mkdir -p $CDDATA/base
mkdir -p $CDDATA/modules
mkdir -p $CDDATA/optional
mkdir -p $CDDATA/devel

echo “copying boot files to $CDDATA…”
cp initrd/$INITRDIMG.gz $CDDATA/initrd.gz
rm initrd/$INITRDIMG.gz
cp -R bootfiles/* $CDDATA
cp -R {info,tools} $CDDATA

# 压缩dir，/etc /var /usr 等
echo “creating compressed images…”

for dir in bin etc home lib opt root usr sbin var; do
if [ -d $ROOT/$dir ]; then
echo “base/$dir.mo”
create_module $ROOT/$dir $CDDATA/base/$dir.mo -keep-as-directory
if [ $? -ne 0 ]; then exit; fi
fi
done

echo “copying kernel from $VMLINUZ…”
cp $VMLINUZ $CDDATA/vmlinuz

制作iso
echo “creating LiveCD ISO image…”
cd $CDDATA
./create_bootiso.sh /tmp/livecd.iso

cd /tmp
header “Your ISO is created in /tmp/livecd.iso”

《星球大战》全集回顾

　　乔治.卢卡斯自编自导的《星球大战》1977年推出立即风靡全球，全球票房高达 7.979亿美元，连电影原声带也创下疯狂销售纪录。接着1980年上映的续集《帝国反击》仍由《星球大战》原班人马担纲演出，影迷反应依旧热烈。 1983年的《绝地归来》加入许多令当时观众大开眼界的计算机动画特效，让星战迷大呼过瘾。完成了《星球大战》系列的后传三部曲之后，乔治.卢卡斯 1996年宣布要开拍前传三部曲。
1999年，《星球大战首部曲》隆重登场；2002年推出的二部曲，全球票房总收入也有6.482亿美元，目前在全球票房收入总排行榜中排在第16名。而《星战》的终结篇将 在明年5月全球公映，相信它的影迷也一定会捧场
